<script setup>
import { useRouter } from 'vitepress'
import { onMounted, ref } from 'vue'

const router = useRouter()
const countdown = ref(10)

onMounted(() => {
  const timer = setInterval(() => {
    countdown.value--
    if (countdown.value <= 0) {
      clearInterval(timer)
      router.go('/')
    }
  }, 1000)
})
</script>
